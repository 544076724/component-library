# Modal 弹框


#### 基本使用说明

<Preview>
  <Story name='basic1'>
      {{
          data(){
              return {
                  visible:false
              }
          },
          methods:{
              handleClose(){
                  this.visible=false
              },
              showModal(){
                   this.visible=true
              }
          },
          components: { CDhnModal },
          template:`<div style="position:relative;">
                <button @click="showModal">展示</button>
                <c-dhn-modal :visible="visible" @handleClose=handleClose>
                    <template #title>
                        title名称
                    </template>
                    <div style="width:300px;height:400px">
                        <h2>这是内容</h2>
                    </div>
                    <template #footer> 这是页脚</template>
                </c-dhn-modal>
              </div>`
      }}
  </Story>
</Preview>

```javascript
<template>
    <div style="position:relative;">
        <button @click="showModal">展示</button>
        <c-dhn-modal :visible="visible" @handleClose=handleClose>
            <template #title>
                title名称
            </template>
            <div style="width:300px;height:400px">
                <h2>这是内容</h2>
            </div>
            <template #footer> 这是页脚</template>
        </c-dhn-modal>
    </div>
</template>
<script>
  export default {
    data(){
        return {
            visible:false
        }
    },
    methods:{
        handleClose(){
            this.visible=false
        },
        showModal(){
            this.visible=true
        }
    },
  }
</script>
```
