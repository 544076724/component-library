# Tabs 标签页

#### 基本使用说明

<Preview>
  <Story name='basic'>
    {{
      components: { CDhnTabs, CDhnTabPane },
      template: `
            <div>
                <c-dhn-tabs v-model="activeName" @tab-click="handleClick">
                    <c-dhn-tab-pane 
                        v-for="item in tabsList" 
                        :label="item.label" 
                        :name="item.name" 
                        :lazy="item.lazy"
                        :key="item.name"
                    >
                            {{item.label}}内容
                    </c-dhn-tab-pane>
                </c-dhn-tabs>
            </div>
      `,
      data () {
        return {
          activeName: 'first',
          tabsList: [
            { label: '用户管理', name: 'first', lazy: 'unInstall' },
            { label: '配置管理', name: 'second', lazy: true }
          ]
        }
      },
      methods: {
        handleClick (tab, event) {
          console.log(tab, event)
        }
      }
    }}
  </Story>
</Preview>
